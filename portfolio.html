pixel-fill.js
// Randomize fill order once on first hover function init() { for (let y = 0; y < canvas.height; y += PIXEL) for (let x = 0; x < canvas.width; x += PIXEL) pixels.push([x, y]); shuffle(pixels); // Fisher-Yates } // Fill ~1/22 of remaining pixels per frame function animate() { const step = Math.ceil(pixels.length / 22); filled = Math.min(pixels.length, filled + direction * step); ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = 'rgba(255,255,255,0.35)'; for (let i = 0; i < filled; i++) ctx.fillRect(...pixels[i], PIXEL-1, PIXEL-1); if ((direction > 0 && filled < pixels.length) || (direction < 0 && filled > 0)) requestAnimationFrame(animate); }
