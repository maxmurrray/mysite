// Shuffle pixel grid once on first hover function init() { for (let y = 0; y < h; y += PIXEL) for (let x = 0; x < w; x += PIXEL) pixels.push([x, y]); // Fisher-Yates shuffle for (let i = pixels.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [pixels[i], pixels[j]] = [pixels[j], pixels[i]]; } } // Fill ~1/22 of squares per frame function animate() { const step = Math.max(1, Math.ceil(pixels.length / 22)); filled = Math.min(pixels.length, Math.max(0, filled + dir * step)); ctx.clearRect(0, 0, w, h); ctx.fillStyle = 'rgba(255,255,255,.35)'; for (let i = 0; i < filled; i++) ctx.fillRect(pixels[i][0], pixels[i][1], PIXEL-1, PIXEL-1); if (dir > 0 ? filled < pixels.length : filled > 0) requestAnimationFrame(animate); }
